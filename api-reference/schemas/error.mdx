Below is a list of error codes that can be returned and what they are used for.

- VALIDATION_ERROR = '100'
- INVALID_RESOURCE = '101'
- FILTER_NOT_ALLOWED = '102'
- INVALID_FIELD_VALUE = '103'
- INVALID_FIELD = '104'
- PARAM_NOT_ALLOWED = '105'
- PARAM_MISSING = '106'
- INVALID_FILTER_VALUE = '107'
- COUNT_MISMATCH = '108'
- KEY_ORDER_MISMATCH = '109'
- KEY_NOT_INCLUDED_IN_URL = '110'
- INVALID_INCLUDE = '112'
- RELATION_EXISTS = '113'
- INVALID_SORT_CRITERIA = '114'
- INVALID_LINKS_OBJECT = '115'
- TYPE_MISMATCH = '116'
- INVALID_PAGE_OBJECT = '117'
- INVALID_PAGE_VALUE = '118'
- INVALID_FIELD_FORMAT = '119'
- INVALID_FILTERS_SYNTAX = '120'
- SAVE_FAILED = '121'
- FORBIDDEN = '403'
- RECORD_NOT_FOUND = '404'
- NOT_ACCEPTABLE = '406'
- UNSUPPORTED_MEDIA_TYPE = '415'
- LOCKED = '423'

<ParamField path="id" type="string">
  a unique identifier for this particular occurrence of the problem. (eg: email#presence)
</ParamField>
<ParamField path="status" type="string">
  the status code applicable to this problem, expressed as a string value.
</ParamField>
<ParamField path="code" type="string">
  a status error code, expressed as a string value.
</ParamField>
<ParamField path="title" type="string">
  a short, human-readable summary of the problem.
</ParamField>
<ParamField path="detail" type="string">
  a human-readable explanation specific to this occurrence of the problem.
</ParamField>
<ParamField path="source" type="string">
  an object containing references to the source of the error via the pointer a JSON Pointer [RFC6901] to the associated entity in the request document [e.g. "/data" for a primary data object, or "/data/attributes/email" for a specific attribute].
</ParamField>
